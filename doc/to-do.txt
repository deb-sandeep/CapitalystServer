CREATE TABLE `capitalyst`.`equity_master` (
  `id` INT NOT NULL,
  `symbol` VARCHAR(45) NOT NULL,
  `company_name` VARCHAR(256) NOT NULL,
  `isin` VARCHAR(45) NOT NULL,
  `industry` VARCHAR(128) NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `symbol_UNIQUE` (`symbol` ASC) VISIBLE,
  UNIQUE INDEX `isin_UNIQUE` (`isin` ASC) VISIBLE);
